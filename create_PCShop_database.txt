CREATE DATABASE IF NOT EXISTS PCShop;
CREATE TABLE IF NOT EXISTS PCShop.clients (user_id INT AUTO_INCREMENT, first_name VARCHAR(45) NOT NULL, family_name VARCHAR(45) NOT NULL, email_address VARCHAR(45) NOT NULL, pass_word VARCHAR(45) NOT NULL, phone_number VARCHAR(45) NOT NULL, money DOUBLE, city VARCHAR(45) NOT NULL, address VARCHAR(45) NOT NULL, CONSTRAINT user_id_PK PRIMARY KEY(user_id));
CREATE TABLE IF NOT EXISTS PCShop.product_types (type_id INT AUTO_INCREMENT, type_name VARCHAR(45) NOT NULL, CONSTRAINT type_id_PK PRIMARY KEY(type_id));
CREATE TABLE IF NOT EXISTS PCShop.products (product_id INT AUTO_INCREMENT, producer_name VARCHAR(45) NOT NULL,model_name VARCHAR(45) NOT NULL,price DOUBLE,product_info MEDIUMTEXT, quantity INT, type_id INT, CONSTRAINT product_id_PK PRIMARY KEY(product_id), CONSTRAINT type_id_FK FOREIGN KEY(type_id) REFERENCES PCShop.product_types(type_id));
CREATE TABLE IF NOT EXISTS PCShop.products_in_carts (user_id INT, product_id INT, quantity_in_cart INT, CONSTRAINT products_in_cart_PK PRIMARY KEY(user_id, product_id), CONSTRAINT type_id_cart_FK FOREIGN KEY(user_id) REFERENCES PCShop.clients(user_id), CONSTRAINT product_id_cart_FK FOREIGN KEY(product_id) REFERENCES PCShop.products(product_id));
CREATE TABLE IF NOT EXISTS PCShop.cases (product_id INT, case_form VARCHAR(45) NOT NULL, case_size VARCHAR(45) NOT NULL, CONSTRAINT product_id_PK PRIMARY KEY(product_id), CONSTRAINT product_id_FK FOREIGN KEY(product_id) REFERENCES PCShop.products(product_id));
CREATE TABLE IF NOT EXISTS PCShop.cpus (product_id INT, number_of_cores INT NOT NULL, clock_speed DOUBLE NOT NULL, socket VARCHAR(45) NOT NULL, CONSTRAINT product_id_cpus_PK PRIMARY KEY(product_id), CONSTRAINT product_i_cpusd_FK FOREIGN KEY(product_id) REFERENCES PCShop.products(product_id));
CREATE TABLE IF NOT EXISTS PCShop.gpus (product_id INT, memory_size VARCHAR(45) NOT NULL, output_interface VARCHAR(45) NOT NULL,max_resolution VARCHAR(45) NOT NULL, CONSTRAINT product_id_gpus_PK PRIMARY KEY(product_id), CONSTRAINT product_id_gpus_FK FOREIGN KEY(product_id) REFERENCES PCShop.products(product_id));
CREATE TABLE IF NOT EXISTS PCShop.hard_drives (product_id INT, drive_size INT NOT NULL, drive_capacity INT NOT NULL,drive_type VARCHAR(45)NOT NULL, CONSTRAINT product_id_hd_PK PRIMARY KEY(product_id), CONSTRAINT product_id_hd_FK FOREIGN KEY(product_id) REFERENCES PCShop.products(product_id));
CREATE TABLE IF NOT EXISTS PCShop.monitors (product_id INT, screen_size DOUBLE NOT NULL, refresh_rate INT NOT NULL, matrix_type VARCHAR(45) NOT NULL, CONSTRAINT product_id_mon_PK PRIMARY KEY(product_id), CONSTRAINT product_id_mon_FK FOREIGN KEY(product_id) REFERENCES PCShop.products(product_id));
CREATE TABLE IF NOT EXISTS PCShop.mother_boards (product_id INT, chipset VARCHAR(45) NOT NULL, ram_slots VARCHAR(45) NOT NULL, socket_type VARCHAR(45) NOT NULL, CONSTRAINT product_id_mb_PK PRIMARY KEY(product_id), CONSTRAINT product_id_mb_FK FOREIGN KEY(product_id) REFERENCES PCShop.products(product_id));
CREATE TABLE IF NOT EXISTS PCShop.rams (product_id INT, ram_type VARCHAR(45) NOT NULL, ram_size INT NOT NULL, CONSTRAINT product_id_ram_PK PRIMARY KEY(product_id), CONSTRAINT product_id_ram_FK FOREIGN KEY(product_id) REFERENCES PCShop.products(product_id));
